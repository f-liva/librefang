```rust
/// Periodically removes image files older than 24 hours from the temp uploads directory.
/// Runs every hour as a background task. (CLEAN-01, CLEAN-02)
async fn cleanup_expired_images() {
    use std::time::{Duration, SystemTime};
    use tokio::time::interval;

    let upload_dir = std::path::PathBuf::from("/tmp/librefang_uploads");
    let max_age = Duration::from_secs(24 * 60 * 60); // 24 hours
    let mut tick = interval(Duration::from_secs(60 * 60)); // every hour

    loop {
        tick.tick().await;

        let entries = match tokio::fs::read_dir(&upload_dir).await {
            Ok(entries) => entries,
            Err(_) => continue, // dir doesn't exist yet, nothing to clean
        };

        let mut removed = 0u32;
        let mut errors = 0u32;
        let mut entries = entries;

        while let Ok(Some(entry)) = entries.next_entry().await {
            let metadata = match entry.metadata().await {
                Ok(m) => m,
                Err(_) => continue,
            };

            if !metadata.is_file() {
                continue;
            }

            let modified = match metadata.modified() {
                Ok(t) => t,
                Err(_) => continue,
            };

            let age = SystemTime::now()
                .duration_since(modified)
                .unwrap_or(Duration::ZERO);

            if age > max_age {
                match tokio::fs::remove_file(entry.path()).await {
                    Ok(()) => removed += 1,
                    Err(e) => {
                        tracing::debug!(
                            path = %entry.path().display(),
                            error = %e,
                            "Failed to remove expired image"
                        );
                        errors += 1;
                    }
                }
            }
        }

        if removed > 0 || errors > 0 {
            tracing::info!(
                removed,
                errors,
                "Image upload cleanup cycle complete"
            );
        }
    }
}
```
